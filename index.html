<script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.2/gsap.min.js"></script>

<script>

  // Set paths for animation

  const blobStart = "M0,0 C100,100 200,100 300,0 L300,400 C200,300 100,300 0,400 Z";

  const blobEnd = "M0,0 C50,150 250,150 300,0 L300,400 C250,250 50,250 0,400 Z";


  // Animate blobs for all members on hover

  for (let i = 1; i <= 4; i++) {

    const member = document.getElementById(`member-${i}`);

    const blob = document.getElementById(`blob-${i}`);


    member.addEventListener('mouseenter', () => {

      gsap.to(blob, {

        duration: 0.6,

        attr: { d: blobEnd },

        ease: "power2.out"

      });

    });


    member.addEventListener('mouseleave', () => {

      gsap.to(blob, {

        duration: 0.6,

        attr: { d: blobStart },

        ease: "power2.in"

      });

    });

  }

</script>
